# Build Docker file
docker build -t node-web-app .

# Start the docker compose 
docker compose up --build -d

# Stop the docker compose
docker compose down
docker compose down -v # Delete old volumes

# Go into the container
docker compose exec web sh

# Web Browser mit der Url starten
start http://localhost

# Delete unused docker resources
docker system prune -a --volumes

# find the 100 largest files in dockerfile
find . -type f -exec du -h {} + | sort -rh | head -n 100

# Print the size of a docker image dir
du -sh
